import { CSSTransition, Transition } from "react-transition-group";
import React, { useEffect, useRef, useState, useCallback } from "react";

interface Props {
  in: boolean;
  delay?: number;
  typewriter?: number;
  children?: React.ReactNode | React.ReactElement | JSX.Element;
}
const TypewriterTransition = (props: Props): JSX.Element => {
  const classes = props.delay
    ? `animate__animated animate__bounce animation_delay-${props.delay}`
    : "animate__animated animate__bounce";

  return (
    <Transition
      in={props.in}
      timeout={{ appear: 1800, enter: 1800, exit: 1800 }}
      mountOnEnter
      unmountOnExit
    >
      {(state) => (
        <div
          style={{ animation: `typewriter 2s steps(${50}) .25s 1 normal both` }}
        >
          {props.children}
        </div>
      )}
    </Transition>
  );
};

export default TypewriterTransition;
