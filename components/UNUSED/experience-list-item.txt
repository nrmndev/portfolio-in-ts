import { Row, Col } from "react-bootstrap";
import type { RootState } from "../../store";
import { useAppSelector } from "../../store/hooks/typedhooks";

type Props = {
  image: string;
  company: string;
  jobposition: string;
  tenureship: string;
  description: string;
  tasks: string[];
  achievements: string[];
  workURL: string;
  workURLText: string;
  message: string;
  onMouseEventHandler: (e: React.MouseEvent<HTMLElement>) => void;
  children?: React.ReactNode;
};
const WorksListItem = (props: Props) => {
  const isActive = useAppSelector(
    (state: RootState) => state.ui.isMessageActive
  );
  const onMouseEventHandler = (
    e: React.MouseEvent<HTMLButtonElement | HTMLDivElement>
  ) => {
    props.onMouseEventHandler(e);
  };
  return (
    <Col sm={12} className={isActive ? "elevate mb-4 mb-md-5" : "mb-4 mb-md-5"}>
      <div className="card overlay">
        <div className="flex-center">
          <Col sm={12} md={4}>
            <img
              src={`../assets/images/${props.image}`}
              alt=""
              className="mr-15p d-block"
              width={225}
              height={107}
            />
          </Col>
          <Col sm={12} md={8}>
            <h3 className="h6 mb-0 color_2">
              Company: <span className="h5 "> {props.company}</span>
            </h3>
            <h4 className="h6 mb-0 color_2">
              Position: <span className="h5"> {props.jobposition}</span>
            </h4>
            <h4 className="h6 color_2 mb-2">
              Tenureship:
              <span className="h5"> {props.tenureship}</span>
            </h4>
          </Col>
        </div>
        <div>
          <p className="color_1 company_info">{props.description}</p>
          <Row>
            <Col lg={6} md={6} sm={12}>
              {props.tasks && (
                <>
                  <h5 className="h6 color_1 mb-0">Tasks:</h5>
                  <ul className="mb-0">
                    {props.tasks.map((e, i) => {
                      return (
                        <li className="color_1" key={i}>
                          {e}
                        </li>
                      );
                    })}
                  </ul>
                </>
              )}
            </Col>
            <Col lg={6} md={6} sm={12}>
              {props.achievements && (
                <>
                  <h4 className="h6 color_1 mb-0">Notable Achievements:</h4>
                  <ul className="mb-0">
                    {props.achievements.map((e, i) => {
                      return (
                        <li className="color_1" key={i}>
                          {e}
                        </li>
                      );
                    })}
                  </ul>
                </>
              )}
            </Col>
          </Row>
        </div>
        <a
          href={props.workURL}
          className="button button__shadowed-light ml-2 z-30"
          target="_blank"
          rel="noreferrer"
        >
          {props.workURLText}
        </a>
        <div
          className="overlay-mask overlay-mask__front"
          onMouseEnter={onMouseEventHandler}
          onMouseLeave={onMouseEventHandler}
        ></div>
      </div>
    </Col>
  );
};

export default WorksListItem;
