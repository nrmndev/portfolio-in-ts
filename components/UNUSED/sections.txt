import { Container, Row, Col } from "react-bootstrap";

type Props = {
  id: string;
  title?: string;
  subTitle?: string;
  subText?: string;
  name?: string;
  className?: string;
  titleClassName?: string;
  subTitleClassName?: string;
  subTextClassName?: string;
  children?: React.ReactNode;
};
const Section = (props: Props): JSX.Element => {
  // const thisSection = useRef();
  const attributes = {
    className: props.className
      ? `${props.className} container-fluid`
      : "container-fluid",
  };
  const titleAttributes = {
    className: props.titleClassName
      ? `${props.titleClassName} text__center text__uppercase font__h1`
      : "text__center text__uppercase font__h1",
  };
  const subTitleAttributes = {
    className: props.subTitleClassName
      ? `${props.subTitleClassName} text-center`
      : "color_filled text-center ",
  };
  const subTextAttributes = {
    className: props.subTextClassName
      ? `${props.subTextClassName} text-center color_1`
      : "text-center color_1",
  };

  const resetHandler: React.MouseEventHandler<HTMLElement> = () => {
    // console.log(e);
  };
  return (
    <section id={props.id} {...attributes} onMouseEnter={resetHandler}>
      <Container>
        <Row className="mb-5 z-2">
          <Col>
            {props.title && <h2 {...titleAttributes}>{props.title}</h2>}
            {props.subTitle && (
              <h4 {...subTitleAttributes}>{props.subTitle}</h4>
            )}
            {props.subText && <p {...subTextAttributes}>{props.subText}</p>}
          </Col>
        </Row>
        {props.children && props.children}
      </Container>
    </section>
  );
};

export default Section;
